## Server interfaces

### IDeviceSettingsServer

The server interface for registering handlers.

```typescript
export interface IDeviceSettingsServer {
  setHandler(handler: IDeviceSettingsHandler): void;
  setHandlerForComponent(
    handler: IDeviceSettingsHandler,
    instance: IComponentInstance,
  ): void;
}
```

### IDeviceSettingsHandler

The handler interface that applications implement.

```typescript
export interface IDeviceSettingsHandler {
  handleGetSetting(query: ISettingQuery): Promise<ISettingResponse>;
  handleUpdateSetting(request: ISettingUpdateRequest): Promise<IUpdateResult>;
  handleListSettings(
    category: SettingCategory,
    filter: ISettingFilter | null,
  ): Promise<ISettingEntry[]>;
}
```

## Response interfaces

### ISettingResponse

```typescript
export interface ISettingResponse {
  getKey(): string;
  getCurrentValue(): string;
  getValueType(): SettingValueType;
  isReadOnly(): boolean;
}
```

### IUpdateResult

```typescript
export interface IUpdateResult {
  getStatus(): UpdateResultStatus;
  getMessage(): string;
}
```

### IUpdateResultBuilder

```typescript
export interface IUpdateResultBuilder {
  updateStatus(status: UpdateResultStatus): IUpdateResultBuilder;
  detailMessage(message: string): IUpdateResultBuilder;
  build(): IUpdateResult;
}
```

<Note>
  `build()` throws `SettingsValidationError` if `updateStatus()` has not been called. The detail message has a maximum length of 8192 characters.
</Note>

## Entry interfaces

### ISettingEntry

```typescript
export interface ISettingEntry {
  getKey(): string;
  getDisplayName(): string;
  getCurrentValue(): string;
  getDefaultValue(): string;
  getValueType(): SettingValueType;
  getCategory(): SettingCategory;
  isReadOnly(): boolean;
  getConstraints(): ISettingConstraints | null;
}
```

### ISettingConstraints

```typescript
export interface ISettingConstraints {
  getMinValue(): number | null;
  getMaxValue(): number | null;
  getStepValue(): number | null;
  getAllowedValues(): string[];
}
```

<Tip>
  Use `ISettingConstraints` to describe valid value ranges for `RANGE` and `ENUM` type settings. For `RANGE`, provide `minValue`, `maxValue`, and optionally `stepValue`. For `ENUM`, provide `allowedValues`.
</Tip>
